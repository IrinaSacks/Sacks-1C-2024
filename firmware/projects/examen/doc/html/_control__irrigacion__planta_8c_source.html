108static void ControlAgua(void *pvParameter){
109 uint8_t estado;
110 while(1){
111 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
112 if (inicio){
113 estado = GPIORead(SENSOR_H);
114 if (estado){
115 GPIOOn(BOMBA_A);
116 agua = true;
117 }
118 else {
119 GPIOOff(BOMBA_A);
120 agua=false;
121 }
122 }
123 }

129static void ControlpH(void *pvParameter){
130 uint16_t lectura_ph;
131 while(1){
132 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
133 if(inicio){
134 AnalogInputReadSingle(CH1, &lectura_ph);
135 nivel_pH = AnalogRaw2mV(lectura_ph)*14/3000;
136
137 if(nivel_pH <6){
138 GPIOOn(BOMBA_pHB);
139 GPIOOff(BOMBA_pHA);
140 }else if(nivel_pH>6.7){
141 GPIOOn(BOMBA_pHA);
142 GPIOOff(BOMBA_pHB);
143 }else if(nivel_pH >=6 && nivel_pH<=6.7){
144 GPIOOff(BOMBA_pHB);
145 GPIOOff(BOMBA_pHA);
146 }
147 }
148 }
149}

154static void MostrarEstado(void *pvParameter){
155 uint16_t mensaje;
156 while(1){
157 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
158 if(inicio){
159 UartSendString(UART_PC, "pH: ");
160 UartSendString(UART_PC, (const char*)UartItoa(nivel_pH, 10));
161
162 if(nivel_pH >= 6 && nivel_pH <=6.7){
163 UartSendString(UART_PC, ", humedad correcta");
164 }else if(nivel_pH >6.7){
165 UartSendString(UART_PC, ", humedad incorrecta");
166 UartSendString(UART_PC, "\r\n");
167 UartSendString(UART_PC, " Bomba de pHA encendida");
168 UartSendString(UART_PC, "\r\n");
169 if(agua){
170 UartSendString(UART_PC, " Bomba de agua encendida");
171 UartSendString(UART_PC, "\r\n");
172 }
173 }else if(nivel_pH<6){
174 UartSendString(UART_PC, ", humedad incorrecta");
175 UartSendString(UART_PC, "\r\n");
176 UartSendString(UART_PC, " Bomba de pHB encendida");
177 UartSendString(UART_PC, "\r\n");
178 if(agua){
179 UartSendString(UART_PC, " Bomba de agua encendida");
180 UartSendString(UART_PC, "\r\n");

182 }
