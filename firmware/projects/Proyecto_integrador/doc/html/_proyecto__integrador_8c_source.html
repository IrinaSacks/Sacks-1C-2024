216static void DetectarLinea(void *pvParameter){
217 while(1){
218 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
219 bool det_izq = GPIORead(gpio_detector_linea[0].pin);
220 bool det_der = GPIORead(gpio_detector_linea[1].pin);
221 if(encendido == true && pausa == false){
222 if(det_izq==true && det_der==true){
223 direccion = 'A';
224 }
225 else if(det_izq==true && det_der==false){
226 direccion = 'D';
227 }
228 else if(det_izq==false && det_der==true){
229 direccion = 'I';
230 }
231 else if(det_izq==false && det_der==false){
232 direccion = 'F';
233 }
234 ControlMotores();
235
236 }else if((encendido==true && pausa==true) || encendido ==false){
237 direccion = 'F';
238 ControlMotores();
239 }else if(encendido==false){
240 direccion = 'F';
241 ControlMotores();
242 }
243
244 }
